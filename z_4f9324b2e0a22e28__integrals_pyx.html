18cdef number_t incgamma(double n, number_t z) nogil: 

19 r""" 

20 incgamma(n, z) 

21  

22 Upper incomplete Gamma function of integer and half-integer degree and real and complex 

23 argument 

24  

25 This function is defined as 

26  

27 .. math 

28  

29 \Gamma(n, z) = \int_z^\infty t^{n - 1} \mathrm e^{-t} \mathrm dt 

30  

31 The negative real axis is the branch cut of the implemented function. 

32  

33 References: 

34 - `DLMF: 8.2 <https://dlmf.nist.gov/8.2>` 

35 """ 

36 cdef long twicen = <long>(2 * n) 

37 if twicen != 2 * n: 

38 raise ValueError(f"{n} is neither integer nor half-integer") 

39 if z == 0: 

40 if twicen <= 0: 

41 return INFINITY 

42 else: 

43 return expd(lgamma(n)) 

44 if twicen == 0: 

45 return cs.exp1(z) 

46 if twicen == 2: 

47 return exp(-z) 

48 if twicen == 1: 

49 return SQPI * cs.erfc(sqrt(z)) 

50 if twicen > 2: 

51 return (n - 1) * incgamma(n - 1, z) + pow(z, n - 1) * exp(-z) 

52 return (incgamma(n + 1, z) - pow(z, n) * exp(-z)) / n

55cdef number_t _intkambe_m3(number_t z, number_t eta) nogil: 

56 r"""Kambe integral of order -3 

57  

58 Implements 

59  

60 .. math:: 

61  

62 I_{-3}(z, \eta) = \sum_{n = 0}^\infty \frac{1}{n!} 

63 \left(\frac{z^2}{4}\right)^{n + 1} 

64 \Gamma(-n - 1, \frac{z^2 \eta^2}{2}) 

65 """ 

66 cdef number_t acc = 0 

67 cdef number_t mult = z * z * 0.25 

68 cdef number_t macc = mult 

69 cdef long mu 

70 for mu in range(21): # TODO: optimize loop 

71 acc += macc * incgamma(-1 - mu, z * z * eta * eta * 0.5) 

72 macc *= mult / (mu + 1) 

73 return acc

97cdef double complex _cintkambe_m2(double_complex z, double complex eta) nogil: 

98 """Kambe integral of order -2 for complex double""" 

99 if creal(z) < 0: 

100 z = -z 

101 cdef double complex faddp = cs.erfc((z * eta - 1j / eta) * M_SQRT1_2) * exp(-1j * z) 

102 cdef double complex faddm = cs.erfc((z * eta + 1j / eta) * M_SQRT1_2) * exp(1j * z) 

103 return -0.5j * (faddp - faddm) * SQPI * M_SQRT1_2

149cdef number_t _intkambe(long n, number_t z, number_t eta, number_t *cache) nogil: 

150 r""" 

151 Cached recursion for the Kambe integral 

152  

153 Args: 

154 n (integer): Order 

155 z (float or complex): Argument 

156 eta (float or complex): Integral cutoff 

157 cache (float or complex, array): Cache of at least size `|n| + 1` 

158  

159 Returns: 

160 float or complex, can return complex values for real argument 

161  

162 References: 

163 .. [#] `K. Kambe, Zeitschrift Fuer Naturforschung A 23, 9 (1968). <https://doi.org/10.1515/zna-1968-0908>`_ 

164 """ 

165 if not isnan(creal(cache[labs(n)])): 

166 return cache[labs(n)] 

167 if eta == 0: 

168 cache[labs(n)] = INFINITY 

169 # n < -4 is handled regularly 

170 elif z == 0 and n == -4: 

171 cache[labs(n)] = -intkambe(-2, 0, eta) + exp(0.5 / (eta * eta)) / eta 

172 elif z == 0 and n == -3: 

173 cache[labs(n)] = exp(1 / (2 * eta * eta)) - 1 

174 # n == -2 is handled regularly 

175 elif z == 0 and n > -2: 

176 cache[labs(n)] = INFINITY 

177 elif n == -3: 

178 cache[labs(n)] = _intkambe_m3(z, eta) 

179 elif n == -2: 

180 cache[labs(n)] = _intkambe_m2(z, eta) 

181 elif n == -1: 

182 cache[labs(n)] = _intkambe_m1(z, eta) 

183 elif n == 0: 

184 cache[labs(n)] = _intkambe_0(z, eta) 

185 # Simplified recursion for n == 1 

186 elif n == 1: 

187 cache[labs(n)] = ( 

188 exp((-z * z * eta * eta + 1 / (eta * eta)) * 0.5) 

189 - _intkambe_m3(z, eta) 

190 ) / (z * z) 

191 # Needed to avoid false cache hits 

192 elif n == 2: 

193 cache[labs(n)] = ( 

194 _intkambe(0, z, eta, cache) 

195 - _intkambe_m2(z, eta) 

196 + eta * exp((-z * z * eta * eta + 1 / (eta * eta)) * 0.5) 

197 ) / (z * z) 

198 # Needed to avoid false cache hits 

199 elif n == 3: 

200 cache[labs(n)] = ( 

201 2 * _intkambe(1, z, eta, cache) 

202 - _intkambe_m1(z, eta) 

203 + eta * eta * exp((-z * z * eta * eta + 1 / (eta * eta)) * 0.5) 

204 ) / (z * z) 

205 elif n < -3: 

206 cache[labs(n)] = ( 

207 (n + 3) * _intkambe(n + 2, z, eta, cache) 

208 - z * z * _intkambe(n + 4, z, eta, cache) 

209 + pow(eta, n + 3) * exp((-z * z * eta * eta + 1 / (eta * eta)) * 0.5) 

210 ) 

211 else: 

212 cache[labs(n)] = ( 

213 (n - 1) * _intkambe(n - 2, z, eta, cache) 

214 - _intkambe(n - 4, z, eta, cache) 

215 + pow(eta, n - 1) * exp((-z * z * eta * eta + 1 / (eta * eta)) * 0.5) 

216 ) / (z * z) 

217 return cache[labs(n)]

220cdef number_t intkambe(long n, number_t z, number_t eta) nogil: 

221 r""" 

222 Integral appearing in the accelerated lattice summations. 

223  

224 Named here after its appearance (in a slightly different form) in equation (3.16), 

225 in Kambe's paper [#]_. 

226  

227 This function is defined as 

228  

229 .. math:: 

230  

231 I_n(\eta, z) 

232 = \int_\eta^\infty t^n \mathrm e^{-\frac{z^2t^2}{2} + \frac{1}{2t^2}} 

233 \mathrm d t 

234  

235 and is calculated via recursion. 

236  

237 Args: 

238 n (integer): Order 

239 z (float or complex): Argument 

240 eta (float or complex): Integral cutoff 

241  

242 Returns: 

243 float or complex 

244  

245 References: 

246 .. [#] `K. Kambe, Zeitschrift Fuer Naturforschung A 23, 9 (1968). <https://doi.org/10.1515/zna-1968-0908>`_ 

247 """ 

248 cdef number_t *cache = <number_t*>malloc((labs(n) + 1) * sizeof(number_t)) 

249 cdef long i 

250 for i in range(labs(n) + 1): 

251 cache[i] = NAN 

252 cdef number_t res = _intkambe(n, z, eta, cache) 

253 free(cache) 

254 return res
